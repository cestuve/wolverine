<script>

let current = 0;

const slides = document.querySelectorAll(".slide");


function nextSlide() {

  slides[current].classList.remove("active");

  current++;

  if (current < slides.length) {

    slides[current].classList.add("active");

  }

}


function answer(correct) {

  slides[current].classList.remove("active");

  current++;


  const text = document.getElementById("resultText");


  if (correct) {

    text.innerHTML = "Correct: fierce, iconic, and difficult to defeat";

  } else {

    text.innerHTML = "We respect your choice, but the correct answer is still Wolverine.";

  }


  slides[current].classList.add("active");

}

</script>
